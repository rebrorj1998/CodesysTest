

FUNCTION MoveRequested : BOOL
VAR_INPUT
	Component : IComponent;
END_VAR
VAR
	movableDevice : IPostHomeMove;
END_VAR

//	Function: MoveIfNeeded()
//	Description: Helper function
//				 Pass in a component - if it has a homable motor, and is not yet homed, start homing
//	Created by: Ben Hess
//	Created Date: May 16th, 2019

IF (__QUERYINTERFACE(Component, movableDevice)) AND Component.OptionActive THEN
	IF NOT movableDevice.InDesiredPosition THEN
		MoveRequested := TRUE;
		RETURN;
	ELSE
		RETURN;
	END_IF
END_IF
